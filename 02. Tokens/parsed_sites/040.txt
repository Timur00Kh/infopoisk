var newsData = null;
var newsHost = '';
var pmData = {
    wrapper: null,
    errorBlock: null
};

function requestNews() {
    var source = newsHost + "?newsproxy",
        xhr = new XMLHttpRequest();

    xhr.open('GET', source, true);
    xhr.onload = function () {
        try { newsData = JSON.parse(xhr.responseText); }
        catch (e) {
            newsData = 'error';
            console.error(e);
        }
    };
    xhr.onerror = function () { newsData = 'error'; };
    xhr.send();
}
function showNews() {
    if (!newsData) setTimeout(showNews, 0);
    else if (newsData == 'error') newsError();
    else {
        var blockBGTypes = [1, 2, 3, 1, 2, 3];

        var newsBlocks = document.createElement('div');
        newsBlocks.className = 'news__blocks swiper-wrapper flexbox';

        for (var i = 0; i < 6; i++) {
            newsBlocks.appendChild(createNewsBlock(newsData[i], blockBGTypes[i]));
        }

        renderNews(newsBlocks);
    }
}
function createNewsBlock(data, typeBG) {
    var link = document.createElement("a");
    link.className = 'news__link flexbox';
    link.href = (data.type === "full" || data.type === "link") ? data.link :
        (data.type === "short") ? newsHost + '_user/user_app.php?mod=news#' + data.id : '';
    link.target = '_blank';
    link.innerHTML = '<div class="head">' +data.date + '</div><div class="body">' + data.label + '</div>';

    var block = document.createElement("div");
    block.className = 'swiper-slide news__block typeBG__' + (typeBG ? typeBG : 1);
    block.appendChild(link);

    return block;
}
function renderNews(content) {
    var spinner = document.getElementById('newsSpinner');
    spinner.classList.add('removing');
    setTimeout(function () {
        spinner.parentNode.removeChild(spinner);

        var newsWrapper = document.getElementById('newsWrapper');
        newsWrapper.classList.remove('prehidden');

        newsWrapper.innerHTML = '';
        var initCarousel = false;
        if (typeof content === 'object' && content.nodeType) {
            newsWrapper.appendChild(content);
            initCarousel = true;
        }
        else {
            newsWrapper.innerHTML = "<div class='news__empty'>Не удалось загрузить новости. Попробуйте позже.</div>";
        }

        newsWrapper.classList.add('showing');
        initCarousel && initCarouselForNews(newsWrapper);
    }, 120);
}
function newsError() {
    renderNews();
}

function initBannersSVGAnim() {
    var containers = [
        {
            selector: '.regBanner',
            path: 'img/redisign_2020/banners/reg.json',
            reserve: '<img src="img/redisign_2020/banners/reg.png">'
        },
        {
            selector: '.wgBanner',
            path: 'img/redisign_2020/banners/wg.json',
            reserve: '<img src="img/redisign_2020/banners/wg.png">'
        },
    ];
    var bannersWrap = document.getElementById('banners');
    containers.forEach(function (svgData) {
        var banners = bannersWrap.querySelectorAll(svgData.selector);
        Array.prototype.forEach.call(banners, function (banner) {
            try {
                var anim = lottie.loadAnimation({
                    container: banner,
                    renderer: 'svg',
                    loop: true,
                    autoplay: true,
                    path: svgData.path
                });
                anim.addEventListener('DOMLoaded', function (e) {
                    anim.wrapper.classList.add('fadeIn');
                })
            }
            catch (err) {
                console.log(err);
                document.getElementById('wgBanner').innerHTML = svgData.reserve;
            }
        });
    });
}
function initPMForm() {
    pmData.wrapper = document.getElementById("pm");
    pmData.errorBlock = pmData.wrapper.querySelector("#errorMsg");
}
function findTeacher(form) {
    pmHideError();
    var xhr = new XMLHttpRequest();

    xhr.onload = function() {
        try {
            var response = JSON.parse(this.responseText);

            if (response.status === 0) pmShowError(response.data);
            else {
                pmData.wrapper.classList.add('showing');
                var payment_form_outer = pmData.wrapper.querySelector("#payment_form_outer");
                payment_form_outer.innerHTML = response.data;
                payment_form_outer.classList.remove("hidden");

                initPMSubForm();
            }
        }
        catch (e) {
            var msg = "Ошибка сервера: Проверьте подключение к интернету или попробуйте позднее.";
            console.error("Техническая информация: " + e);
            pmShowError(msg);
        }
    };
    xhr.open (form.method, form.action, true);
    xhr.setRequestHeader("X-Requested-With", 'XMLHttpRequest');
    xhr.send(new FormData(form));

    return false;
}
function recommendSum(price) {
    var payment_sum = pmData.wrapper.querySelector("#payment_sum");
    payment_sum.value = price;
    payment_sum.disabled = false;
}
function pmShowError(error) {
    pmData.errorBlock.innerHTML = error;
    pmData.errorBlock.classList.add('showing');
}
function pmHideError() {
    pmData.errorBlock.classList.remove('showing');
}
function initPMSubForm() {
    var contestInput = document.getElementById('ridx');
    contestInput.onkeypress = function (event) {
        event.preventDefault();
        return;
    }
    contestInput.onclick = function () {
        var isShowing = this.dataset.listshowing === '0' ? false : true;
        openCloseContestList(contestList, !isShowing);
        this.dataset.listshowing = !isShowing ? '1' : '0'
    }

    var contestList = document.getElementById('ridxList');
    contestList.onclick = function (event) {
        var target = event.target;
        while (target != this) {
            if (target.dataset.hasOwnProperty('price') && target.dataset.hasOwnProperty('value')) {
                document.getElementById('ridxHidden').value = target.dataset.value;
                contestInput.value = target.innerText;
                recommendSum(target.dataset.price);
                break;
            }
            target = target.parentNode;
        }

        contestInput.dataset.listshowing = "0";
        openCloseContestList(this, false);
    }

    document.addEventListener('click', function(e) {
        var target = e.target;
        if (target != contestList && target != contestInput) {
            contestInput.dataset.listshowing = "0";
            openCloseContestList(this, false);
        }
    });
}
function openCloseContestList(contestList, show) {
    document.getElementById('choose_contest').classList[show ? 'add' : 'remove']('showing');
    document.getElementById('ridxList').classList[show ? 'add' : 'remove']('showing');
}
